1.edit etc/hosts for ubuntu to include

127.0.0.1 localhost
127.0.0.1 ubuntu.ubuntu-domain ubuntu

2.
wget http://www.gtlib.gatech.edu/pub/apache/hbase/1.1.0.1/hbase-1.1.0.1-bin.tar.gz
tar xzvf hbase-1.1.0.1-bin.tar.gz
cd hbase-1.1.0.1

3.Edit conf/hbase-env.sh

export JAVA_HOME=/usr
export HBASE_MANAGES_ZK=false #if you want to use existing zk cluster. default is true, in which case HBase will start its own ZK cluster

4.Edit conf/hbase-site.xml

  	<property>
                <name>hbase.rootdir</name>
                <value>file:///home/testuser/hbase</value> <!--will be an hdfs:// address if deployed on cluster-->
        </property>

        <property>
                <name>hbase.zookeeper.property.dataDir</name>
                <value>/home/testuser/zookeeper</value>
        </property>

        <property>
                <name>hbase.cluster.distributed</name>
                <value>true</value>  <!--if you jps , you can see different processes spawned by hbase-->
        </property>

	<property>
                        <name>hbase.zookeeper.property.clientPort</name>
                        <value>2181</value> <!--2181 is default ZK port-->
	</property>	

<property>
      <name>hbase.zookeeper.quorum</name>
      <value>wirthmaster</value> <!--comma separated , default localhost-->
</property>



5. conf/regionservers
list of region servers, format same to slave file

6. conf/backup-masters

--------
sample commands

may need to force HDFS out of safe mode

sudo -i
bin/hadoop dfsadmin -safemode leave

./bin/start-hbase.sh

./bin/hbase shell



create 'table_name', 'col_family'

list 'table_name'

put 'table_name', 'row1', 'col_family:col_name', 'value1'

scan 'table_name'

get 'table_name', 'row1'

disable 'table_name'

drop 'table_name'

5.
check hdfs folder

./bin/hadoop fs -ls /hbase

can start up to 10 HMaster servers

Each HMaster uses 16010, 16020, and 16030

#start backup HMasters with backup, number is port offset
./bin/local-master-backup.sh 2 3 5

#kill a backup master, number is port offset
cat /tmp/hbase-testuser-1-master.pid |xargs kill -9

#start local region servers
.bin/local-regionservers.sh start 2 3 4 5

#stop a single local region server
.bin/local-regionservers.sh stop 3
